	// 이중배열
	var arr = [];
	for (var i = 0; i < 5; i++){
		arr[i] = [] ; // 5개의 배열이 들어감
		for (var j = 0; j < 5; j++){
			 arr[i][j] = i * j;
		}
	}

	// 랜덤
	for (var i = 0; i < 5; i++){
		arr[i] = Math.floor(Math.random() * 7);	// 7까지의 수가 랜덤으로 나옴

	}
